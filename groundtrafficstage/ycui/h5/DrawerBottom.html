    # 使用

        :show="show"
        @close="show = false"
        :maskShow="maskShow"
        :canTouchMove="true"
      >
        <div style="height: 200px;padding: 40px">
            Content...
        </div>
  </yc-drawer-bottom>

    # 属性详情

    参数 说明 类型 可选值 默认值
    show 显隐状态控制 Boolean — false
    closeBtnClass 关闭按钮样式 String — —
    maskShow 是否显示图层 Boolean — false
    drawerBottomEvent 弹窗显隐回调 Function — —
    canTouchMove 是否可滚动 Boolean — true
    closeIcon 关闭按钮图标 String — —
    closeBtnImg 关闭按钮背景图 String — false
    showClose 显示关闭按钮 Boolean — true
    specialIpt 是否存在兼容ios异步input Boolean — false
    titleShow 是否显示标题 Boolean — true
    isHandleIphoneX 是否处理苹果X 底部空隙 Boolean — false
    background 背景色 String — #fff
    borderRadius 是否显示圆角 Boolean — false
    isShowBottomSlide 键盘做处理 Boolean — false
    maxHeight 最大高度 String — —
